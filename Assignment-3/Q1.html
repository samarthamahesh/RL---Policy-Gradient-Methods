class GridWorld():

    def __init__(self, size, gamma, noise):
        self.actions = [(1, 0), (0, 1), (-1, 0), (0, -1)]
        self.n_actions = len(self.actions)
        self.n_states = size*size
        self.size = size
        self.noise = noise
        self.gamma = gamma

        self.transition_matrix = np.zeros((self.n_states, self.n_actions, self.n_states))
        for i in range(self.n_states):
            for j in range(self.n_actions):
                for k in range(self.n_states):
                    self.transition_matrix[i][j][k] = self.get_transition_prob(i, j, k)

    def ind_pt(self, ind):
        return ind % self.size, ind // self.size

    def pt_ind(self, x, y):
        return y * self.size + x

    def is_neighbor(self, s1, s2):
        if np.abs(s1[0]-s2[0])+np.abs(s1[1]-s2[1]) <= 1:
            return True
        return False

    def get_transition_prob(self, state, action, next_state):
        x1, y1 = self.ind_pt(state)
        dx, dy = self.actions[action]
        x2, y2 = self.ind_pt(next_state)

        if not self.is_neighbor((x1, y2), (x2, y2)):
            return 0

        if (x1+dx, y1+dy) == (x2, y2):
            return 1 - self.noise + self.noise / self.n_actions

        if (x1, y1) != (x2, y2):
            return self.noise / self.n_actions

        if (x1, y1) in [(0, 0), (0, self.size-1), (self.size-1, 0), (self.size-1, self.size-1)]:

            if (x1 + dx < 0 or x1 + dx >= self.size) or (y1 + dy < 0 or y1 + dy >= self.size):
                return 1 - self.noise + 2*self.noise/self.n_actions

            else:
                return 2*self.noise/self.n_actions

        else:
            
            if x1 not in [0, self.size-1] and y1 not in [0, self.size-1]:
                return 0

            if (x1 + dx < 0 or x1 + dx >= self.size) or (y1 + dy < 0 or y1 + dy >= self.size):
                return 1 - self.noise + self.noise/self.n_actions

            else:
                return self.noise/self.n_actions

def irl(env, policy, Rmax, lamda):

    nS = env.n_states
    nA = env.n_actions
    P = env.transition_matrix

    A = set(range(nA))

    c = -np.hstack([np.zeros(nS), np.ones(nS), -lamda * np.ones(nS)])

    A_mat = np.zeros((0, nS))
    tmp1 = np.zeros((0, nS))
    for s in range(nS):
        for a in A - {policy[s]}:
            A_mat = np.vstack((A_mat, -np.dot(P[s, policy[s]] - P[s, a], np.linalg.inv(np.eye(nS) - env.gamma * P[:, policy[s]]))))
            tmp1 = np.vstack((tmp1, np.eye(1, nS, s)))

    A_mat = np.vstack((A_mat, A_mat, -np.eye(nS), np.eye(nS)))
    tmp1 = np.vstack((tmp1, np.zeros((nS * (nA-1), nS)), np.zeros((nS, nS)), np.zeros((nS, nS))))
    tmp2 = np.vstack((np.zeros((nS*(nA-1), nS)), np.zeros((nS*(nA-1), nS)), -np.eye(nS), -np.eye(nS)))

    A_mat = np.hstack((A_mat, tmp1, tmp2))

    tmp = np.hstack((np.vstack((-np.eye(nS), np.eye(nS))), np.vstack((np.zeros((nS, nS)), np.zeros((nS, nS)))), np.vstack((np.zeros((nS, nS)), np.zeros((nS, nS))))))
    A_mat = np.vstack((A_mat, tmp))

    b = np.vstack((np.zeros((2*nS*nA, 1)), Rmax * np.ones((2 * nS, 1))))

    A_mat = matrix(A_mat)
    b = matrix(b)
    c = matrix(c)

    results = solvers.lp(c, A_mat, b)
    r = np.asarray(results["x"][:nS], dtype=np.double)

    return r.reshape(nS)

Analysis¶

I have prepared a class named GridWorld that models the noisy nxn sized grid MDP with states, actions, transition probabilities, discount factor. However, We dont know the reward here which is the main thing we are trying to extract using IRL. However, we are given the optimal policy using which we approximate the reward function for each state. The optimal policy I have used is taken from the paper. The groundtruth reward function is 0 reward for all other states and Rmax reward for the terminal state [cell (5, 5)].

After running the experiments using different regularization (lamda), we see that with no reguarization (lamda = 0), we see lot of noise in the reward function and it is quite far from the groundtruth rewards. But with regularization (lamda = 1.05), we see that the reward function is very near to the groundtruth reward function, with all rewards being closer to 0. As the optimal reward for terminal state is Rmax, I have set to Rmax after LPP is solved in both cases.

Hence, we see that with regularization, te constrints are satisfied to larger extent and hence we see that the optimized reward function comes near to graoundtruth reward function.
